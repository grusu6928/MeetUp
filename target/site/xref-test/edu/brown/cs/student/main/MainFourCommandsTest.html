144   @Test
145   public void testSortList() {
146     setUp();
147 
148     List<String> starOneData = Arrays.asList("1", "Star One", "5.0", "4.0", "6.0");
149     List<String> starTwoData = Arrays.asList("2", "Star Two", "5.0", "4.0", "6.0");
150     List<String> starThreeData = Arrays.asList("3", "Star Three", "5.0", "4.0", "6.0");
151     List<String> starFourData = Arrays.asList("4", "Star Four", "5.0", "4.0", "6.0");
152 
153     Star star1 = new Star(starOneData);
154     Star star2 = new Star(starTwoData);
155     Star star3 = new Star(starThreeData);
156     Star star4 = new Star(starFourData);
157 
158     star1.setDistance(10.0);
159     star2.setDistance(8.0);
160     star3.setDistance(25.0);
161     star4.setDistance(11.0);
162 
163     ArrayList<Star> starList = new ArrayList<>();
164     starList.add(star1);
165     starList.add(star2);
166     starList.add(star3);
167     starList.add(star4);
168 
169     ArrayList<Star> preSortedList = new ArrayList<>();
170 
171     for (int i = 0; i < starList.size(); i++) {
172       preSortedList.add(starList.get(i));
173     }
174 
175     mainFourCommands.sortList(starList, new EuclideanDistComparator(1));
176 
177     assertTrue(starList.get(0).getDistance() == preSortedList.get(1).getDistance());
178     assertTrue(starList.get(1).getDistance() == preSortedList.get(0).getDistance());
179     assertTrue(starList.get(2).getDistance() == preSortedList.get(3).getDistance());
180     assertTrue(starList.get(3).getDistance() == preSortedList.get(2).getDistance());
181 
182     tearDown();
183 
184   }
