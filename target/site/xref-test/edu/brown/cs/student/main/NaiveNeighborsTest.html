66    @Test
67    public void testChooseRandomly() {
68      setUp();
69  
70      List<String> starOneData = Arrays.asList("1", "Star One", "5.0", "4.0", "6.0");
71      List<String> starTwoData = Arrays.asList("2", "Star Two", "5.0", "4.0", "6.0");
72      List<String> starThreeData = Arrays.asList("3", "Star Three", "5.0", "4.0", "6.0");
73      List<String> starFourData = Arrays.asList("4", "Star Four", "5.0", "4.0", "6.0");
74  
75      Star star1 = new Star(starOneData);
76      Star star2 = new Star(starTwoData);
77      Star star3 = new Star(starThreeData);
78      Star star4 = new Star(starFourData);
79  
80      // size of 4
81      ArrayList<Star> equalitySet = new ArrayList<>();
82      equalitySet.add(star1);
83      equalitySet.add(star2);
84      equalitySet.add(star3);
85      equalitySet.add(star4);
86  
87      ArrayList<Star> forSureSet = stars;
88  
89      Random r = new Random();
90  
91      int counter = 0;
92      while (counter < 2) {
93        counter++;
94  
95        Integer min = 0;
96        Integer max = stars.size() - 1;
97        Integer rand = r.nextInt((max - min) + 1) + min;
98        forSureSet.remove(rand);
99      }
100 
101     int forSureSetSize = forSureSet.size();
102 
103     int numToSelect = 2;
104 
105     ArrayList<Star> finalSet = naiveNeighborsCommand.chooseRandomly(equalitySet, numToSelect, forSureSet);
106 
107     assertTrue(finalSet.size() == forSureSetSize + numToSelect);
