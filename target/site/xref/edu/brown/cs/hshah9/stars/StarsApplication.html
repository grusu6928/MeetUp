34    /**
35     * For the GUI, prints out stars.
36     * @param starsList list of stars
37     * @return stars list to print out
38     */
39    public String getOutput(ArrayList<Star> starsList) {
40      String output = "";
41      for (int i = 0; i < starsList.size(); i++) {
42        output = output +  (i + 1) + ". " + starsList.get(i).getName() + "\n";
43      }
44      return output;
45    }
46  
47    /**
48     * Runs the Star's program, using a HashMap to go from inputted command to action.
49     * @param input caller input
50     */
51    public void run(String[] input) {
52      String command = input[0];
53  
54      if (!validCommand(commandsMap, command) && !command.equals("")) {
55        System.out.println("ERROR: Invalid command entered");
56      } else {
57        Set<String> setOfCommands = commandsMap.keySet();
58        Iterator<String> it = setOfCommands.iterator();
59  
60        // DESIGN CHOICE: ALL COMMANDS PART OF COMMANDS INTERFACE
61        // SO CAN ALL BE CALLED THE SAME WAY (VIA EXECUTE COMMAND)
62        while (it.hasNext()) {
63          if (it.next().equals(command)) {
64            CommandManager commandInstance = commandsMap.get(command);
65            commandInstance.execute(input);
66          }
67        }
68      }
69  
70    }
71  
72    private HashMap<String, CommandManager> createCommandsMap() {
73      HashMap<String, CommandManager> commands = new HashMap<>();
74      commands.put("stars", new StarsCommand());
75      commands.put("naive_neighbors", new NaiveNeighborsCommand());
76      commands.put("neighbors", new NeighborsCommand());
77      commands.put("naive_radius", new NaiveRadiusCommand());
78      commands.put("radius", new RadiusCommand());
79      commands.put("mock", new MockCommand());
80  
81      return commands;
82    }
83  
84    private boolean validCommand(HashMap<String, CommandManager> setOfCommands, String command) {
85      return setOfCommands.keySet().contains(command);
86    }
