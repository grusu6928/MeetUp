3   import edu.brown.cs.hshah9.stars.HasCoordinates;
4   
5   import java.util.ArrayList;
6   import java.util.Collections;
7   import java.util.List;
8   
9   
10  /**
11   * This class represents a KDTree, which is comprised of KDNodes whose values are objects
12   * implementing the HasCoordinates interface.
13   * @param <T> object that implements HasCoordinates interface
14   */
15  public final class KDTree<T extends HasCoordinates> {
16  
17    private KDNode<T> root;
18    private int currDim;
19  
20  
21    /**
22     * Instantiates KDTree from a list.
23     * @param data list from which to make tree
24     * @param totalDims dimensions of tree
25     * @param startingDim first dimension to sort by
26     */
27    public KDTree(ArrayList<T> data, int totalDims, int startingDim) {
28      currDim = startingDim;
29      Collections.sort(data, new NodeComparator(currDim)); // sort by 1st dimension -> X
30      root = this.addNode(data, currDim, totalDims); // first recursive call
31    }
32  
33    /**
34     * Adds a node to the KDTree. This is a recursive function. Note that in the case of equality,
35     * it is not specified whether a node will be added to the left or right subtree.
36     * @param data list of nodes from which to select the node to add
37     * @param dim current iteration's dimension of coordinate comparison
38     * @param totalDims total number of dimensions the KDNode objects have
39     * @return added node
40     */
41    public KDNode<T> addNode(List<T> data, Integer dim, Integer totalDims) {
42  
43      // base case (out of data)
44      if (data.size() == 0) {
45        return null;
46      }
47  
48      // new node is in middle of sorted list
49      int mid = data.size() / 2;
50      KDNode<T> node = new KDNode(data.get(mid), null, null);
51  
52      // increment dimension before recursion
53      dim = (dim + 1) % totalDims;
54  
55      // create left and right sublists
56      List<T> leftSublist = data.subList(0, mid);
57      List<T> rightSublist = data.subList(mid + 1, data.size());
58  
59      // sort sublists
60      Collections.sort(leftSublist, new NodeComparator(dim));
61      Collections.sort(rightSublist, new NodeComparator(dim));
62  
63      // recursive calls
64      node.setLeftChild(this.addNode(leftSublist, dim, totalDims));
65      node.setRightChild(this.addNode(rightSublist, dim, totalDims));
66  
67      return node;
68    }
69  
70    /**
71     * Gets the root of the KDTree, which is useful for traversals.
72     * @return root of tree
73     */
74    public KDNode<T> getRoot() {
75      return root;
76    }
77  }
